python -m venv myvenv - This will create a virtual environment named myvenv

python -m venv myvenv python=3.8 - This will create a virtual environment named myvenv and install python3.8
conda create -n tensorflow_env python=3.9 - Git Bash

myvenv\Scripts\activate  - This will activate the environment

source myvenv/Scripts/activate - When using Git Bash

python -m pip install --upgrade pip  - This will upgrade pip

pip install -r requirements.txt  - This will install libraries listed in the requirements.txt file

Ctrl + Shift + P  - To select the correct python interpreter
